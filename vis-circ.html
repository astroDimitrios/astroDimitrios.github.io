import numpy as np import matplotlib.pyplot as plt import mpl_toolkits.axes_grid1 from js import setInterval, clearInterval from pyodide import to_js out = Element("outputDiv") help = Element("outputHelp") import datetime def update_timestamp(elem): elem.write( format(datetime.datetime.now(), "%H:%M:%S, %A, %d %B %Y") ) class Play_Buttons(): def __init__(self): pass def start(self): self.tm = setInterval( to_js(lambda: update_timestamp(out)), 1000 # Update every 1000 ms ) def stop(self): clearInterval(self.tm) player = Play_Buttons() def main_play(*ags, **kws): player.start() def main_play_back(*ags, **kws): help.write('Here') def main_stop(*ags, **kws): player.stop()
